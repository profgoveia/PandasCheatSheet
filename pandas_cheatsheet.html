Nulos · Duplicatas · Drop

# Remover nulos
df.dropna()
df.dropna(subset=["col1", "col2"])
df.dropna(how="all", axis=1)  # col 100% vazia

# Preencher nulos
df["Loja"].fillna("Online")
df["Renda"].fillna(df["Renda"].median())
df.ffill()   # propaga valor anterior
df.bfill()   # propaga valor seguinte

# Remover duplicatas
df.drop_duplicates(subset=["ID_Pedido"], keep="last")

# Remover colunas
df.drop(columns=["Col1", "Col2"])
LimpezaPrefira fillna(mediana) com outliers presentes
✏️

Texto · Tipos · Datas

# Padronização encadeada
df["Loja"] = (df["Loja"]
    .fillna("Online").str.strip().str.title())

# Extrair com regex
df["DDD"] = df["Fone"].str.extract(r"\((\d{2})\)")

# Converter tipos
df["Valor"] = df["Valor"].astype(float)
df["Cat"]   = df["Cat"].astype("category") # -90% RAM

# Datas
df["Data"] = pd.to_datetime(df["Data"], format="%Y-%m-%d")
df["Ano"]  = df["Data"].dt.year
df["Dia"]  = df["Data"].dt.day_name()
df["MesAno"] = df["Data"].dt.to_period("M")

# Interpolação de nulos em série temporal
df["Vendas"] = df["Vendas"].interpolate(method="time")

🔢

clip · replace · where · mask

# Limitar outliers entre min e max
df["Idade"] = df["Idade"].clip(0, 120)

# Substituir valores com dicionário
df["Status"] = df["Status"].replace({"S":"Sim", "N":"Não"})

# where: mantém onde condição é True
df["Valor"] = df["Valor"].where(df["Valor"] > 0, other=0)

# mask: substitui onde condição é True
df["Score"] = df["Score"].mask(df["Score"] > 100, 100)

# Detectar outliers com IQR
Q1, Q3 = df["Val"].quantile([.25, .75])
IQR = Q3 - Q1
out = df[(df["Val"] < Q1-1.5*IQR) | (df["Val"] > Q3+1.5*IQR)]
Outliersclip() é mais simples do que filtrar e deletar

groupby() + agg()

# Soma simples
df.groupby("Loja")["Fat"].sum()

# Manter coluna (não virar índice)
df.groupby("Loja", as_index=False)["Fat"].sum()

# Múltiplas agregações
df.groupby("Produto").agg({
    "Faturamento": ["sum", "mean", "max"],
    "Qtd": "sum",
})

# Agregação nomeada (pandas ≥ 0.25)
df.groupby("Loja").agg(
    Fat_Total    = ("Fat", "sum"),
    Ticket_Medio = ("Fat", "mean"),
    Qtd_Pedidos  = ("ID", "count"),
)
AnáliseAgregação nomeada gera colunas com nome limpo
🪟

transform() · rolling() · expanding()

Agrega sem reduzir o número de linhas — essencial para criar features.

# transform: mantém o shape original
df["Fat_Media_Loja"] = (
    df.groupby("Loja")["Fat"].transform("mean")
)

# Desvio do valor em relação à média do grupo
df["Desvio"] = (
    df["Fat"] -
    df.groupby("Loja")["Fat"].transform("mean")
)

# Média móvel de 3 períodos
df["MM3"] = df["Vendas"].rolling(3).mean()

# Faturamento acumulado
df["Fat_Acum"] = df["Fat"].expanding().sum()

# Rank dentro do grupo
df["Rank"] = (
    df.groupby("Loja")["Fat"]
      .rank(ascending=False)
)
Feature Engineeringtransform() é mais rápido que apply() para isso
⬆️

sort_values() · nlargest() · rank()

# Ordenação simples / múltipla
df.sort_values("Fat", ascending=False)
df.sort_values(["Região", "Fat"],
               ascending=[True, False])

# Top N / Bottom N
df.nlargest(10, "Faturamento")
df.nsmallest(5, "Ticket_Medio")

# Ranking global (método dense = sem gaps)
df["Rank"] = df["Fat"].rank(

# Redefinir índice após sort/groupby
df.reset_index(drop=True)

7 · Feature Engineering — criação de variáveis
🧮

get_dummies() — One-Hot Encoding

Converte variáveis categóricas em colunas binárias 0/1. Obrigatório para ML.

# Encoding básico
df_enc = pd.get_dummies(df, columns=["Loja", "Produto"])

# drop_first=True evita multicolinearidade
df_enc = pd.get_dummies(df, columns=["Loja"],
                          drop_first=True, dtype=int)

# Tratar NaN como categoria própria
df_enc = pd.get_dummies(df, dummy_na=True)

# Alternativa: LabelEncoder do sklearn
from sklearn.preprocessing import LabelEncoder
df["Cat_Cod"] = LabelEncoder().fit_transform(df["Cat"])
ML Prepdrop_first=True obrigatório em regressão linear
✂️

cut() · qcut() — Discretização

Transforma variáveis contínuas em categorias (bins).

# cut: intervalos manuais
df["Faixa_Etaria"] = pd.cut(
    df["Idade"],
    bins=[0, 18, 35, 60, 120],
    labels=["Jovem", "Adulto", "Sênior", "Idoso"]
)

# qcut: quantis (igual frequência)
df["Quartil"] = pd.qcut(
    df["Faturamento"], q=4,
    labels=["Q1", "Q2", "Q3", "Q4"]
)

# Decis de score de risco (segmentação)
df["Decil"] = pd.qcut(
    df["Score"], q=10,
    labels=range(1, 11)
)
Segmentaçãoqcut garante grupos com igual nº de observações
⚡

np.where · np.select · apply

# np.where: condição binária
df["Bateu Meta"] = np.where(
    df["Fat"] >= df["Meta"], "Sim", "Não"
)

# np.select: múltiplas condições
conds   = [df["Score"] >= 80, df["Score"] >= 60]
choices = ["Alto", "Médio"]
df["Perfil"] = np.select(conds, choices, default="Baixo")

# apply: função customizada por linha
df["Margem"] = df.apply(
    lambda r: r["Lucro"] / r["Receita"], axis=1
)

# map: mapeamento via dicionário
df["Num"] = df["Status"].map({"Ativo":1, "Inativo":0})
Feature Eng.np.select() substitui múltiplos np.where encadeados
↕️

shift() · diff() · pct_change()

Cria variáveis de lag, diferença e variação — essenciais em séries temporais.

# Valor do período anterior (lag 1)
df["Vendas_Lag1"] = df["Vendas"].shift(1)

# Lag de 12 meses (comparação ano a ano)
df["Vendas_AA"] = df["Vendas"].shift(12)

# Diferença absoluta
df["Variacao"] = df["Vendas"].diff(1)

# Variação percentual
df["Crescimento"] = df["Vendas"].pct_change()

# Dentro de grupos
df["Var_Loja"] = (
    df.groupby("Loja")["Vendas"].pct_change()
)
Séries Temporaisshift() dentro de groupby evita data leakage

col = df["Valor"]
df["Norm"] = (col-col.min()) / (col.max()-col.min())

# Log transform (reduz skewness)
df["LogValor"] = np.log1p(df["Valor"])

Do CSV ao modelo: pipeline real

import pandas as pd
import numpy  as np

# 1. Ingestão
df = pd.read_csv("clientes.csv", encoding="latin1", parse_dates=["Data_Cadastro"])

# 2. Inspeção rápida
print(df.shape, df.isnull().sum().to_dict())

# 3. Limpeza
df = df.drop_duplicates(subset=["ID"]).dropna(subset=["Valor"])
df["Renda"]  = df["Renda"].fillna(df["Renda"].median())
df["Cidade"] = df["Cidade"].str.strip().str.title()
df["Valor"]  = df["Valor"].clip(0, df["Valor"].quantile(0.99))

# 4. Feature engineering
df["Dias_Cliente"] = (pd.Timestamp.today() - df["Data_Cadastro"]).dt.days
df["Log_Renda"]   = np.log1p(df["Renda"])
df["Faixa_Renda"] = pd.qcut(df["Renda"], 4, labels=["D","C","B","A"])
df["Perfil"]      = np.select(
    [df["Score"] >= 80, df["Score"] >= 60],
    ["Alto", "Médio"], default="Baixo"
)

# 5. Encoding
df_enc = pd.get_dummies(df,
    columns=["Cidade", "Faixa_Renda", "Perfil"],
    drop_first=True, dtype=int
)

# 6. Export
df_enc.to_parquet("clientes_processados.parquet", index=False)
